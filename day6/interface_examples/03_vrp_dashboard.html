int: n_customers = 10; int: n_vehicles = 3; int: capacity = 100; set of int: CUSTOMERS = 1..n_customers; set of int: VEHICLES = 0..n_vehicles-1; % 0 = depot array[0..n_customers, 0..n_customers] of int: dist; array[CUSTOMERS] of int: demand; % ── Variables ────────────────────────────────────────────────── % next[i] = node จาก i ไป (0 = depot) array[0..n_customers] of var 0..n_customers: next; array[0..n_customers] of var VEHICLES: vehicle; array[VEHICLES] of var 0..capacity: load; % ── Constraints ──────────────────────────────────────────────── % ทุกลูกค้าถูกเยี่ยมครั้งเดียว include "alldifferent.mzn"; constraint alldifferent(next); % capacity ของแต่ละคัน constraint forall(v in VEHICLES)( load[v] = sum(c in CUSTOMERS)(demand[c] * bool2int(vehicle[c] = v)) ); constraint forall(v in VEHICLES)(load[v] <= capacity); % ── Objective ────────────────────────────────────────────────── var int: total_dist = sum(i in 0..n_customers)(dist[i, next[i]]); solve minimize total_dist;
